# 校园二手交易平台 - 部署指南

## 环境准备

### 1. 安装 Node.js

```bash
# 使用 nvm 安装（推荐）
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.0/install.sh | bash
nvm install 18
nvm use 18

# 验证安装
node -v
npm -v
```

### 2. 安装 JDK 17

```bash
# Ubuntu/Debian
sudo apt update
sudo apt install openjdk-17-jdk

# CentOS/RHEL
sudo yum install java-17-openjdk

# 验证安装
java -version
```

### 3. 安装 MySQL 8.0

```bash
# Ubuntu/Debian
sudo apt install mysql-server
sudo systemctl start mysql
sudo systemctl enable mysql

# macOS
brew install mysql
brew services start mysql

# 验证安装
mysql --version
```

### 4. 安装 Maven

```bash
# Ubuntu/Debian
sudo apt install maven

# macOS
brew install maven

# 验证安装
mvn -version
```

### 5. 安装 Redis（可选）

```bash
# Ubuntu/Debian
sudo apt install redis-server
sudo systemctl start redis
sudo systemctl enable redis

# macOS
brew install redis
brew services start redis

# 验证安装
redis-cli ping
```

## 项目部署

### 方式一：开发环境部署

#### 1. 初始化数据库

```bash
# 登录MySQL
mysql -u root -p

# 创建数据库
CREATE DATABASE campus_trade DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

# 使用数据库
USE campus_trade;

# 执行初始化脚本
SOURCE /path/to/campus-second-hand/sql/init.sql;

# 验证数据
SHOW TABLES;
SELECT * FROM user;
```

#### 2. 配置后端

```bash
# 编辑配置文件
vim backend/src/main/resources/application.yml

# 修改数据库配置
spring:
  datasource:
    url: jdbc:mysql://localhost:3306/campus_trade?useUnicode=true&characterEncoding=utf-8&useSSL=false&serverTimezone=Asia/Shanghai
    username: root
    password: your_password

# 修改Redis配置（如未安装Redis，注释掉redis相关配置）
```

#### 3. 启动后端

```bash
cd backend

# 清理并编译
mvn clean package -DskipTests

# 启动应用
mvn spring-boot:run
# 或
java -jar target/campus-second-hand-1.0.0.jar
```

后端启动成功后会显示：
```
Started CampusTradeApplication in X seconds
```

#### 4. 启动前端

```bash
cd frontend

# 安装依赖
npm install

# 启动开发服务器
npm run dev
```

前端启动成功后会显示：
```
VITE v5.x.x ready in XXX ms

➜  Local: http://localhost:3000/
➜  Network: http://192.168.x.x:3000/
```

#### 5. 访问系统

- 打开浏览器访问：http://localhost:3000
- 登录后台管理：http://localhost:3000/admin/dashboard
- 默认管理员账号：`admin` / `admin123`

### 方式二：生产环境部署

#### 1. 构建前端

```bash
cd frontend

# 安装依赖
npm install

# 构建生产版本
npm run build

# 构建产物在 dist 目录
ls -la dist/
```

#### 2. 配置 Nginx

```bash
# 安装 Nginx
sudo apt install nginx

# 编辑 Nginx 配置
sudo vim /etc/nginx/sites-available/campus-trade
```

```nginx
server {
    listen 80;
    server_name your_domain.com;

    # 前端静态文件
    location / {
        root /var/www/campus-trade/dist;
        index index.html;
        try_files $uri $uri/ /index.html;
    }

    # API 代理
    location /api {
        proxy_pass http://127.0.0.1:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    # 上传文件
    location /uploads {
        alias /var/www/campus-trade/uploads;
    }
}
```

```bash
# 启用站点配置
sudo ln -s /etc/nginx/sites-available/campus-trade /etc/nginx/sites-enabled/

# 测试配置
sudo nginx -t

# 重启 Nginx
sudo systemctl restart nginx

# 复制前端构建产物
sudo cp -r frontend/dist/* /var/www/campus-trade/
```

#### 3. 构建后端

```bash
cd backend

# 清理并构建
mvn clean package -DskipTests

# 构建产物在 target 目录
ls -la target/*.jar
```

#### 4. 配置 MySQL

```sql
-- 创建生产数据库
CREATE DATABASE campus_trade_prod DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 创建应用数据库用户
CREATE USER 'campus'@'localhost' IDENTIFIED BY 'StrongPassword123!';
GRANT ALL PRIVILEGES ON campus_trade_prod.* TO 'campus'@'localhost';
FLUSH PRIVILEGES;

-- 导入初始数据
USE campus_trade_prod;
SOURCE /path/to/campus-second-hand/sql/init.sql;
```

#### 5. 配置后端应用

```bash
# 创建应用目录
sudo mkdir -p /opt/campus-trade

# 复制 jar 包
sudo cp target/campus-second-hand-1.0.0.jar /opt/campus-trade/

# 创建配置文件
sudo vim /opt/campus-trade/application-prod.yml
```

```yaml
server:
  port: 8080

spring:
  datasource:
    driver-class-name: com.mysql.cj.jdbc.Driver
    url: jdbc:mysql://localhost:3306/campus_trade_prod?useUnicode=true&characterEncoding=utf-8&useSSL=true&serverTimezone=Asia/Shanghai
    username: campus
    password: StrongPassword123!
  redis:
    host: localhost
    port: 6379

jwt:
  secret: your-very-long-secret-key-for-production
  expiration: 86400000
```

#### 6. 创建 Systemd 服务

```bash
sudo vim /etc/systemd/system/campus-trade.service
```

```ini
[Unit]
Description=Campus Second Hand Trade Platform
After=network.target mysql.service redis.service

[Service]
Type=simple
User=www-data
WorkingDirectory=/opt/campus-trade
ExecStart=/usr/bin/java -jar campus-second-hand-1.0.0.jar --spring.config.location=application-prod.yml
Restart=on-failure
RestartSec=10

[Install]
WantedBy=multi-user.target
```

```bash
# 重新加载 systemd
sudo systemctl daemon-reload

# 启动服务
sudo systemctl start campus-trade

# 设置开机自启
sudo systemctl enable campus-trade

# 查看服务状态
sudo systemctl status campus-trade
```

#### 7. 配置 SSL（可选）

```bash
# 安装 Certbot
sudo apt install certbot python3-certbot-nginx

# 获取 SSL 证书
sudo certbot --nginx -d your_domain.com

# 自动续期测试
sudo certbot renew --dry-run
```

### Docker 部署（推荐）

#### 1. 创建 Docker Compose 文件

```bash
vim docker-compose.yml
```

```yaml
version: '3.8'

services:
  mysql:
    image: mysql:8.0
    container_name: campus-mysql
    environment:
      MYSQL_ROOT_PASSWORD: rootpassword
      MYSQL_DATABASE: campus_trade
      MYSQL_USER: campus
      MYSQL_PASSWORD: campus123
    ports:
      - "3306:3306"
    volumes:
      - mysql_data:/var/lib/mysql
      - ./sql/init.sql:/docker-entrypoint-initdb.d/init.sql
    command: --default-authentication-plugin=mysql_native_password

  redis:
    image: redis:7-alpine
    container_name: campus-redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data

  backend:
    build: ./backend
    container_name: campus-backend
    ports:
      - "8080:8080"
    depends_on:
      - mysql
      - redis
    environment:
      SPRING_DATASOURCE_URL: jdbc:mysql://mysql:3306/campus_trade?useUnicode=true&characterEncoding=utf-8&useSSL=false&serverTimezone=Asia/Shanghai
      SPRING_DATASOURCE_USERNAME: root
      SPRING_DATASOURCE_PASSWORD: rootpassword
      SPRING_REDIS_HOST: redis
    volumes:
      - ./uploads:/uploads

  frontend:
    image: nginx:alpine
    container_name: campus-frontend
    ports:
      - "80:80"
    volumes:
      - ./frontend/dist:/usr/share/nginx/html
      - ./nginx.conf:/etc/nginx/conf.d/default.conf
    depends_on:
      - backend

volumes:
  mysql_data:
  redis_data:
```

#### 2. 创建 Nginx 配置

```bash
vim nginx.conf
```

```nginx
server {
    listen 80;
    server_name localhost;
    root /usr/share/nginx/html;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }

    location /api {
        proxy_pass http://backend:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
}
```

#### 3. 构建并启动

```bash
# 构建前端
cd frontend && npm install && npm run build && cd ..

# 启动所有服务
docker-compose up -d

# 查看服务状态
docker-compose ps

# 查看日志
docker-compose logs -f
```

## 常见问题

### 1. 端口被占用

```bash
# 查找占用端口的进程
lsof -i :3000
lsof -i :8080

# 终止进程
kill -9 <PID>
```

### 2. 数据库连接失败

```bash
# 检查MySQL服务状态
sudo systemctl status mysql

# 检查MySQL连接
mysql -u root -p -h localhost

# 检查防火墙
sudo ufw status
sudo ufw allow 3306
```

### 3. 前端构建失败

```bash
# 清除缓存
rm -rf node_modules package-lock.json
npm cache clean --force
npm install

# 检查Node版本
node -v  # 需要 >= 16.0
```

### 4. 后端启动失败

```bash
# 查看详细日志
mvn spring-boot:run -X

# 检查配置文件
cat src/main/resources/application.yml
```

### 5. 文件上传失败

```bash
# 创建上传目录
sudo mkdir -p /uploads
sudo chmod 777 /uploads

# 检查Nginx配置
sudo nginx -t
```

sudo tail -f /var/log/nginx/access.log

```bash
# 备份
mysqldump -u root -p campus_trade > backup_$(date +%Y%m%d).sql

# 恢复
mysql -u root -p campus_trade < backup_20240101.sql
```
